    **Goal:** When a player walks through a door, instead of firing the scene transition locally, route it through the server so all players move together.

    **1. Find the existing door/room-transition handler** in game-scene.ts. It likely calls `this.scene.start(SCENE_KEYS.GAME_SCENE, { levelName: ..., spawnX: ..., spawnY: ... })` or similar. Name it `_handleDoorTransition` or whatever it currently is.

    **2. Wrap it with an online-mode check:**
    ```typescript
    private _handleDoorTransition(targetRoom: string, spawnX: number, spawnY: number): void {
      // Check if networked session is active
      let nm: NetworkManager | null = null;
      try { nm = NetworkManager.getInstance(); } catch { /* offline */ }

      if (nm && nm.isConnected) {
        // Online: request server to broadcast to all clients
        nm.sendRoomTransitionRequest({ targetRoom, spawnX, spawnY });
        // Do NOT transition locally — wait for server echo
      } else {
        // Offline: fire immediately
        this._doRoomTransition(targetRoom, spawnX, spawnY);
      }
    }

    private _doRoomTransition(targetRoom: string, spawnX: number, spawnY: number): void {
      // Move existing scene restart logic here
      this.scene.restart({ levelName: targetRoom, spawnX, spawnY });
    }
    ```

    **3. Register `NETWORK_ROOM_TRANSITION` listener:**
    In `create()` (or `init()`), after the NetworkManager guard:
    ```typescript
    try {
      const nm = NetworkManager.getInstance();
      if (nm.isConnected) {
        EVENT_BUS.on(CUSTOM_EVENTS.NETWORK_ROOM_TRANSITION, this._onNetworkRoomTransition, this);
      }
    } catch { /* offline, skip */ }
    ```

    ```typescript
    private _onNetworkRoomTransition(payload: RoomTransitionPayload): void {
      this._doRoomTransition(payload.targetRoom, payload.spawnX, payload.spawnY);
    }
    ```

    **4. Cleanup in `shutdown()`:**
    ```typescript
    EVENT_BUS.off(CUSTOM_EVENTS.NETWORK_ROOM_TRANSITION, this._onNetworkRoomTransition, this);
    ```

    **IMPORTANT:** Do not change any offline game logic. All changes are additive with guard clauses.

    **`_onRemotePlayerUpdate(payload: PlayerUpdateBroadcast)`:**
    ```typescript
    private _onRemotePlayerUpdate(payload: PlayerUpdateBroadcast): void {
      const nm = NetworkManager.getInstance();
      // Don't apply our own update
      if (payload.playerId === nm.localPlayerId) return;

      let remote = this.#remotePlayers.get(payload.playerId);
      if (!remote) {
        // Spawn new remote player
        const slotIndex = this.#remotePlayers.size + 1;  // 0 reserved for local
        const tint = PLAYER_TINT_PALETTE[slotIndex % PLAYER_TINT_PALETTE.length];
        remote = new Player({
          scene: this,
          x: payload.x,
          y: payload.y,
          playerId: payload.playerId,
          tintColor: tint,
          // copy other required PlayerConfig defaults from local player config
        });
        this.#remotePlayers.set(payload.playerId, remote);
      }

      // Apply position snapshot (interpolate toward target)
      remote.x = Phaser.Math.Linear(remote.x, payload.x, 0.3);
      remote.y = Phaser.Math.Linear(remote.y, payload.y, 0.3);

      // Apply remote input snapshot so animations play correctly
      const ric = remote.inputComponent as RemoteInputComponent;
      if (ric && typeof ric.applySnapshot === 'function') {
        ric.applySnapshot({ direction: payload.direction, currentState: payload.currentState });
      }
    }
    ```

    **`_onRemotePlayerDisconnected(payload: PlayerDisconnectedPayload)`:**
    ```typescript
    private _onRemotePlayerDisconnected(payload: PlayerDisconnectedPayload): void {
      const remote = this.#remotePlayers.get(payload.playerId);
      if (remote) {
        remote.destroy();
        this.#remotePlayers.delete(payload.playerId);
      }
      // Show disconnect message for 3 seconds
      const name = payload.playerName ?? 'A player';
      const msg = this.add.text(
        this.cameras.main.centerX,
        this.cameras.main.centerY - 40,
        `${name} disconnected`,
        { fontFamily: '"Press Start 2P"', fontSize: '8px', color: '#ff4444' }
      ).setOrigin(0.5).setScrollFactor(0).setDepth(999);
      this.time.delayedCall(3000, () => msg.destroy());
    }
    ```

    this.#remotePlayers.forEach(p => p.destroy());
