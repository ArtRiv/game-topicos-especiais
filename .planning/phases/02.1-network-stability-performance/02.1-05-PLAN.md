<objective>
Add server-side tick batching and grid-based spatial interest management to the relay pipeline, reducing per-message overhead and scaling bandwidth sub-linearly with player count.

Purpose: Without batching, the server forwards each position message individually — at 20 players × 20 Hz = 400 messages/sec inbound, each fanned out to 19 clients = 7,600 outbound messages/sec. Tick batching aggregates all position updates received within a 50ms window into one outbound message per client. Spatial interest management further reduces this by throttling updates for distant players.

Output: Server accumulates position updates per tick and sends one batched message per client per tick, with distant players throttled to ~5 Hz.
</objective>

    ```typescript
    export class SpatialGrid {
      #cellSize: number;
      #playerPositions = new Map<string, { x: number; y: number; cellKey: string }>();
      #nearbyThreshold: number; // cells — players within this many cells are "nearby"

      constructor(cellSize = 256, nearbyThreshold = 2) {
        this.#cellSize = cellSize;
        this.#nearbyThreshold = nearbyThreshold;
      }

      /** Update a player's position in the grid */
      updatePosition(playerId: string, x: number, y: number): void {
        const cellKey = this.#getCellKey(x, y);
        this.#playerPositions.set(playerId, { x, y, cellKey });
      }

      /** Remove player from grid */
      removePlayer(playerId: string): void {
        this.#playerPositions.delete(playerId);
      }

      /** Check if two players are within the nearby threshold */
      areNearby(playerA: string, playerB: string): boolean {
        const a = this.#playerPositions.get(playerA);
        const b = this.#playerPositions.get(playerB);
        if (!a || !b) return true; // unknown position → send updates (safe default)
        const [ax, ay] = this.#getCellCoords(a.x, a.y);
        const [bx, by] = this.#getCellCoords(b.x, b.y);
        return Math.abs(ax - bx) <= this.#nearbyThreshold && Math.abs(ay - by) <= this.#nearbyThreshold;
      }

      #getCellKey(x: number, y: number): string {
        const [cx, cy] = this.#getCellCoords(x, y);
        return `${cx},${cy}`;
      }

      #getCellCoords(x: number, y: number): [number, number] {
        return [Math.floor(x / this.#cellSize), Math.floor(y / this.#cellSize)];
      }
    }
    ```

    2. Cell size 256px is a good default for a Phaser top-down game (roughly one screen width at typical zoom). `nearbyThreshold = 2` means players within 2 cells (~512px) are "nearby."

    ```typescript
    import { encode } from '@msgpack/msgpack';

    type PendingUpdate = {
      playerId: string;
      data: Record<string, unknown>; // decoded position message with playerId
    };

    export class TickBatcher {
      #tickRateHz: number;
      #tickInterval: ReturnType<typeof setInterval> | null = null;
      // Per tick, accumulate latest position per player (only latest matters)
      #pendingPositions = new Map<string, Record<string, unknown>>(); // playerId → latest position msg
      #flushCallback: ((batch: Map<string, Record<string, unknown>>) => void) | null = null;

      constructor(tickRateHz = 20) {
        this.#tickRateHz = tickRateHz;
      }

      /** Start the tick loop. flushCallback receives the accumulated positions map each tick. */
      start(flushCallback: (batch: Map<string, Record<string, unknown>>) => void): void {
        this.#flushCallback = flushCallback;
        const intervalMs = Math.round(1000 / this.#tickRateHz);
        this.#tickInterval = setInterval(() => this.#flush(), intervalMs);
      }

      stop(): void {
        if (this.#tickInterval) {
          clearInterval(this.#tickInterval);
          this.#tickInterval = null;
        }
        this.#pendingPositions.clear();
      }

      /** Queue a position update. Only the latest per player is kept (overwrites previous). */
      queuePosition(playerId: string, message: Record<string, unknown>): void {
        this.#pendingPositions.set(playerId, message);
      }

      #flush(): void {
        if (this.#pendingPositions.size === 0 || !this.#flushCallback) return;
        // Pass the batch to the relay for fan-out
        this.#flushCallback(new Map(this.#pendingPositions));
        this.#pendingPositions.clear();
      }
    }
    ```

    2. Key design: Only the LATEST position per player is kept per tick. If a player sends 2 updates in one 50ms window, only the most recent is forwarded. This naturally deduplicates.

       ```typescript
       #onTickFlush(batch: Map<string, Record<string, unknown>>): void {
         // For each room, build a batched message array for each recipient
         // Group updates by room
         for (const [senderPlayerId, posMsg] of batch) {
           const senderSocketId = this.#playerIdToSocketId(senderPlayerId);
           if (!senderSocketId) continue;
           const lobbyId = this.#socketToLobbyId.get(senderSocketId);
           if (!lobbyId) continue;
           const room = this.#gameRooms.get(lobbyId);
           if (!room) continue;

           const otherSocketIds = room.getOtherSocketIds(senderSocketId);
           for (const recipientSocketId of otherSocketIds) {
             const recipientPlayerId = room.getPlayerIdBySocketId(recipientSocketId);
             if (!recipientPlayerId) continue;

             // Spatial interest management
             const nearby = this.#spatialGrid.areNearby(senderPlayerId, recipientPlayerId);
             if (!nearby) {
               // Throttle distant updates: send every 4th tick (~5 Hz at 20 Hz tick rate)
               const key = `${senderPlayerId}:${recipientPlayerId}`;
               const count = (this.#distantTickCounter.get(key) ?? 0) + 1;
               this.#distantTickCounter.set(key, count);
               if (count % 4 !== 0) continue; // skip 3 out of 4 ticks
             }

             // Send to recipient's unreliable channel
             const peer = this.#peers.get(recipientSocketId);
             if (peer?.unreliable) {
               peer.unreliable.sendMessageBinary(encode(posMsg));
             }
           }
         }
       }
       ```
