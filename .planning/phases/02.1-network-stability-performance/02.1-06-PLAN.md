       ```typescript
       #playerHealth = new Map<string, number>(); // playerId → current HP
       static readonly MAX_HEALTH = 6; // matches client's PLAYER_START_MAX_HEALTH

       addPlayer(playerId: string, socketId: string): void {
         this.#players.set(socketId, playerId);
         this.#playerHealth.set(playerId, GameRoom.MAX_HEALTH);
       }

       removePlayer(socketId: string): string | undefined {
         const playerId = this.#players.get(socketId);
         if (playerId) this.#playerHealth.delete(playerId);
         this.#players.delete(socketId);
         return playerId;
       }

       /** Returns new HP after damage, or null if target not found or already dead */
       applyDamage(targetPlayerId: string, damage: number): { newHp: number; eliminated: boolean } | null {
         const hp = this.#playerHealth.get(targetPlayerId);
         if (hp === undefined || hp <= 0) return null;
         const newHp = Math.max(0, hp - damage);
         this.#playerHealth.set(targetPlayerId, newHp);
         return { newHp, eliminated: newHp <= 0 };
       }

       getPlayerHealth(playerId: string): number | undefined {
         return this.#playerHealth.get(playerId);
       }

       isPlayerAlive(playerId: string): boolean {
         const hp = this.#playerHealth.get(playerId);
         return hp !== undefined && hp > 0;
       }
       ```

    4. **Add damage handling to WebRTC relay** (`game-server/src/webrtc-relay.ts`):
       In the reliable channel message handler, add a case for `type: 'damage'`:
       ```typescript
       case 'damage': {
         // Validate: is attacker alive? is target alive? is target in same room?
         const targetPlayerId = msg.targetPlayerId;
         if (!room.isPlayerAlive(senderPlayerId)) break; // dead players can't deal damage
         if (!room.isPlayerAlive(targetPlayerId)) break; // dead target — ignore
         const result = room.applyDamage(targetPlayerId, msg.damage);
         if (!result) break;

         // Broadcast damage-confirmed to ALL players in room (including attacker for feedback)
         const confirmed = { type: 'damage-confirmed', targetPlayerId, damage: msg.damage, newHp: result.newHp, attackerPlayerId: senderPlayerId };
         this.#broadcastReliableToRoom(lobbyId, confirmed);

         // If eliminated, broadcast separately
         if (result.eliminated) {
           const eliminated = { type: 'player-eliminated', playerId: targetPlayerId, eliminatedByPlayerId: senderPlayerId };
           this.#broadcastReliableToRoom(lobbyId, eliminated);
         }
         break;
       }
       ```

    5. **Add `#broadcastReliableToRoom` helper** to webrtc-relay.ts:
       ```typescript
       #broadcastReliableToRoom(lobbyId: string, message: Record<string, unknown>): void {
         const room = this.#gameRooms.get(lobbyId);
         if (!room) return;
         const encoded = encode(message);
         for (const [socketId] of this.#peers) {
           // Only send to peers in this room
           if (this.#socketToLobbyId.get(socketId) !== lobbyId) continue;
           const peer = this.#peers.get(socketId);
           if (peer?.reliable) {
             peer.reliable.sendMessageBinary(encoded);
           }
         }
       }
       ```
